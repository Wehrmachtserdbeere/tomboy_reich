			base = 10

					is_in_faction_with = ENG

				add = 75

					is_guaranteed_by = ENG
					is_guaranteed_by = FRA
					is_guaranteed_by = GER
				}
			}
		}

			add_opinion_modifier = {

			}
		}

		days = 1
	}

		set_global_flag = BEL_liege_massacre_happened

				add_manpower = -500000
			}
		}
	}

		name = tomboy.9.a
		
		trigger = {
			NOT = {
				tag = BEL
				has_war_with = BEL
			}
		}

        		limit = { NOT = { has_war_with = BEL } }

		            target = PREV

		        }
    		}
		}
	}
